## Instrucciones

***********************************************
				Instalación
***********************************************
LINUX:

$ sudo apt-get update

$ sudo apt-get install python3-pip python3-dev nginx

$ sudo pip3 install virtualenv

$ apt-get install python3-venv

virtualenv flaskprojectenv

cd $(nombreWeb)

$ python3 -m venv venv

$ virtualenv venv

$ source venv/bin/activate

(vent)$ pip install flask

***********************************************
				 Ejecución
***********************************************
$ source venv/bin/activate
	(linux)
-------------------------
$ venv\Scripts\activate
	(windows)

(vent)$ flask run
